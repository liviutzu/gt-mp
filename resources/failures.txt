        public static int getPlayerMoney(string target)
        {
            int money;
            string stringMySQLQuery = "SELECT * FROM `users` WHERE `username` = '" + target + "'";
            MySqlDataReader readerMySQL = new MySqlCommand(stringMySQLQuery, mysqlHandler.connectionMySQL).ExecuteReader();

            readerMySQL.Read();
            money = readerMySQL.GetInt32("bani");
            readerMySQL.Close();

            return money;
        }
